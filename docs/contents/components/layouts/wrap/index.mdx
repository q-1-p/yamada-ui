---
title: Wrap
description: >-
  `Wrap` is a component that has `wrap` set on `Flex`. It inherits convenient
  style shorthand from `Flex`.
package_name: "@yamada-ui/layouts"
order: 5
with_description: true
---

## Import

```ts
import { Wrap } from "@yamada-ui/react"
```

## Usage

```tsx
<Wrap gap="md">
  <For each={Array.from({ length: 20 })}>
    {(_, index) => (
      <For key={index} each={["primary", "secondary", "warning", "danger"]}>
        {(bg, nestedIndex) => (
          <Box
            key={`${index}-${nestedIndex}`}
            p="md"
            rounded="md"
            bg={bg}
            color="white"
          >
            Box
          </Box>
        )}
      </For>
    )}
  </For>
</Wrap>
```
